create table user(
	user_id	int	auto_increment PRIMARY KEY,
	user_act	char(20)	not null,
	constraint user_act_unique unique(user_act),
	user_pwd	char(20)	not null,
	user_name	char(20)	not null,
	user_email	char(30)	not null,
	constraint user_email_unique unique(user_email),
	user_number	char(30)	not null,
	constraint user_number_unique unique(user_number),
	user_id_number	char(30),
	constraint user_id_number_unique unique(user_id_number),
	user_state	tinyint	not null,
	department_id	int	not null,
	role_id	int	not null
	);
